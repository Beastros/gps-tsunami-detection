function esc(s){return String(s==null?'':s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/"/g,'&quot;');}
var REASON_PLAIN={no_coherent_pairs:'No tsunami-like pattern between GPS stations',insufficient_stations:'Not enough GPS stations had usable data',kp_gated:'Geomagnetic storm — GPS check paused',no_pairs:'No qualifying station pairs in the time window'};
function plainReason(code,fallback){if(!code)return fallback||'Analysis finished';var k=String(code).toLowerCase();return REASON_PLAIN[k]||fallback||'Analysis finished';}
function stationListUpper(ev){var a=ev.rinex_acquired_stations||(ev.prediction&&ev.prediction.stations_processed)||[];return a.map(function(s){return String(s).toUpperCase();});}
function awaitingArchiveNote(ev){var w=ev.rinex_awaiting_archive||[];if(!w.length)return '';var show=w.slice(0,6).map(function(s){return String(s).toUpperCase();}).join(', ');if(w.length>6)show+=' +'+(w.length-6);return 'Still waiting on public GPS files: '+show+'.';}
function humanTec(ev){var det=!!(ev.algo_detected||ev.prediction&&ev.prediction.detected);var gated=!!ev.space_weather_gated;if(det)return{label:'Signal seen',badge:'ok',tip:'GPS found a coherent tsunami-like ionospheric pattern.'};if(gated)return{label:'Storm hold',badge:'gated',tip:'Space weather was too disturbed for a reliable GPS check.'};var reason=ev.detection_reason||(ev.prediction&&ev.prediction.reason);var st=stationListUpper(ev);var tip=plainReason(reason,'No tsunami-like GPS pattern was found.');if(st.length)tip+=' Stations analyzed: '+st.join(', ')+'.';if(ev.data_status==='partial'||ev.data_status==='awaiting_archive'){var a=awaitingArchiveNote(ev);if(a)tip+=' '+a;}return{label:'No signal',badge:'silent',tip:tip};}
function humanOutcome(ev){var det=!!ev.algo_detected;var o=ev.outcome||'';if(o==='TRUE_POSITIVE')return{label:'Confirmed',tip:'GPS and tide gauges agree a tsunami-like signal was present.',tone:'good'};if(o==='TRUE_NEGATIVE'||o==='CORRECT_ABSTENTION_NO_ANCHOR')return{label:'Quiet ocean',tip:'No tsunami-like signal in GPS or tide gauges.',tone:'neutral'};if(o==='FALSE_POSITIVE')return{label:'GPS only',tip:'GPS flagged a pattern but tide gauges did not confirm.',tone:'bad'};if(o==='FALSE_NEGATIVE'){if(!det&&ev.any_gauge_tsunami)return{label:'Gauges only',tip:'Small tide-gauge motion; GPS did not see a matching ionospheric signal. Not a live tsunami alert.',tone:'warn'};return{label:'Missed',tip:'Tide gauges saw something GPS did not pick up.',tone:'warn'};}if(o==='GEOMETRY_LIMITED_MISS')return{label:'Out of range',tip:'Station geometry limits which sites can contribute.',tone:'neutral'};return{label:'—',tip:o||'Not scored yet',tone:'neutral'};}
function humanDart(ds){if(ds==='confirmed')return{sym:'\u25CF',tip:'Ocean buoy saw a pressure signal',col:'var(--accent)'};if(ds==='negative')return{sym:'\u25CB',tip:'Buoy data — no tsunami-like pressure',col:'var(--dim)'};if(ds==='pending')return{sym:'~',tip:'Buoy check pending',col:'var(--dim)'};if(ds==='no_buoys')return{sym:'—',tip:'No buoy in range for this event',col:'var(--dim)'};return{sym:'—',tip:'Buoy data not available',col:'var(--dim)'};}
function humanNearMiss(reason){var r=(reason||'').toLowerCase();if(r.indexOf('below threshold')>=0)return 'Below M6.5 threshold';if(r.indexOf('too deep')>=0)return 'Too deep';if(r.indexOf('strike')>=0||r.indexOf('rake')>=0||r.indexOf('thrust')>=0)return 'Focal mechanism filter';if(r.indexOf('outside')>=0||r.indexOf('pacific')>=0)return 'Outside Pacific watch';return reason||'Did not queue';}
function describeLiveEvent(ev){var st=(ev.status||'queued').toLowerCase();var mw='Mw'+(ev.magnitude!=null?(+ev.magnitude).toFixed(1):'?');var place=ev.place||'Unknown location';var zone=ev.zones&&ev.zones.length?ev.zones[0]:'Pacific';var head=mw+' · '+place;var pred=ev.prediction||{};var awaitNote=awaitingArchiveNote(ev);if(st==='queued'||st==='pending_rinex'){var sub='On watch list. GPS files are downloaded a few hours after the earthquake.';if(awaitNote)sub+=' '+awaitNote;return{tone:'info',title:'AWAITING DATA — '+zone.toUpperCase(),detail:head,tag:'Queued',note:sub};}if(st==='rinex_failed'){return{tone:'warn',title:'AWAITING GPS FILES — '+zone.toUpperCase(),detail:head,tag:(ev.rinex_retries?'Retry '+ev.rinex_retries:'Archive backlog'),note:(awaitNote||'Public GPS archives do not have enough files yet.')+' Automatic retries continue. Not a coast warning.'};}if(st==='detector_failed'){return{tone:'error',title:'ANALYSIS PAUSED — '+zone.toUpperCase(),detail:head,tag:'Will retry',note:'A processing step failed; the pipeline will try again. Not a tsunami warning.'};}if(st==='rinex_ready'){return{tone:'info',title:'ANALYZING GPS — '+zone.toUpperCase(),detail:head,tag:'In progress',note:'Running the ionospheric detector on downloaded GPS data.'+(awaitNote?' '+awaitNote:'')};}if(st==='predicted'){if(pred.detected){return{tone:'alert',title:'POSSIBLE IONOSPHERIC SIGNAL — '+zone.toUpperCase(),detail:head,tag:'Research alert',note:'GPS pattern matches tsunami-like criteria. Not an official warning — confirm with NOAA and local authorities.'};}var stations=stationListUpper(ev);var note=plainReason(pred.reason,'Analysis complete — no tsunami-like GPS pattern.');if(stations.length)note+=' Stations used: '+stations.join(', ')+'.';if(awaitNote)note+=' '+awaitNote;if(ev.retro_trigger_reason)note+=' Re-checking because new GPS files appeared on the archive.';return{tone:'info',title:'NO GPS TSUNAMI SIGNATURE — '+zone.toUpperCase(),detail:head,tag:'Analysis complete',note:note};}if(ev.data_status==='awaiting_archive'||ev.data_status==='partial'){return{tone:'warn',title:'PARTIAL GPS DATA — '+zone.toUpperCase(),detail:head,tag:'Awaiting archive',note:(awaitNote||'Some nearby GPS files are not on the public archive yet.')+' Monitoring continues.'};}return{tone:'info',title:'MONITORING — '+zone.toUpperCase(),detail:head,tag:'In progress',note:'Pipeline is processing this event.'};}
function applyBannerTone(banner,tone){banner.classList.remove('tone-alert','tone-warn','tone-error','tone-info');if(tone)banner.classList.add('tone-'+tone);var dot=banner.querySelector('[data-alert-dot]');var titleEl=document.getElementById('alert-title');var col=tone==='alert'||tone==='error'?'#ff4a4a':tone==='warn'?'#ffa726':tone==='info'?'var(--accent)':'var(--text-hi)';if(dot)dot.style.background=col;if(titleEl)titleEl.style.color=tone==='info'?'var(--accent)':tone==='warn'?'#ffa726':tone==='alert'||tone==='error'?'#ff4a4a':'var(--text-hi)';}
